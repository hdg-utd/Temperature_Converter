Problem: Convert the temperature to Celsius, Fahrenheit or Kelvin when one of the above is givenInput: celsius temp, fahrenheit temp, kelvin tempOutput: celsius temp, fahrenheit temp, kelvin tempget celsius tempset fahrenheit temp = (9/5)*celsius temp + 32set kelvin temp = celsius temp + 273.15display fahrenheit temp, kelvin tempget fahrenheit tempset celsius temp = (5/9)*(celsius temp - 32)set kelvin temp = celsius temp + 273.15display celsius temp, kelvin tempget kelvin tempset celsius temp = kelvin temp - 273.15set fahrenheit temp = (9/5)*celsius temp + 32display celsius temp, fahrenheit temphalt
